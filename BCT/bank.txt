pragma solidity 0.8.18;

contract bank{

    int balance;

    constructor(){
        balance = 50;
    }

    function getBalance() view public returns(int){
        return balance;
    }

    function deposit(int amount) public{
        balance = balance + amount;
    }
    
    function withdraw(int amount) public{
        balance = balance - amount;
    }
}
